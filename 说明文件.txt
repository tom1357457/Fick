typedef struct DLinkNode
{
	int sum = 0;//车站总数                                                  
	int number = 0;//新添的序列号
	string subname;//新添的地铁名
	int population = 0;
	struct DLinkNode* prior;
	struct DLinkNode* next;
}DLinkNode;
